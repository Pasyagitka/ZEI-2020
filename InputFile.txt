symbolic func  	 foo(symbolic str)
[
set symbolic mystr : "max"!
set tiny p    : symblen(str)!
giveback mystr!
]
perform 
[
~set symbolic r : foo(r)!
set tiny x : 0!
set tiny y : 010!

y : x / 2!
set tiny a : x+y+2!
set symbolic 	str : "sp ace"!
set symbolic  mystr : "а бв?гдежзикла	мопростуфхцч"!
set logical mb : false!                                                                    ~алащуащуа
when (x ^ y)
[
~foo()!
set tiny m!
show("string")!
~show()!
]
otherwise
[
set tiny nb : symbtotiny(mystr)!
~show(nb)!
]
set tiny z :  (x * y)	* ( x #y)!

loop(z < 5)
[
    z : z+1!
]
~show(z)!
]
